<script>

    function calcolaFrequenza() {

        // Recupero i dati

        const LC = parseFloat(document.getElementById('lc').value);

        const OP = parseFloat(document.getElementById('op').value);

        const PP = parseFloat(document.getElementById('pp').value);

       

        const spanPercentuale = document.getElementById('percentuale');

        const divMessaggio = document.getElementById('messaggio');

 

        // Validazione minima

        if (isNaN(LC) || isNaN(OP) || isNaN(PP) || LC <= 0) {

            divMessaggio.innerText = "Per favore, compila tutti i campi correttamente.";

            divMessaggio.className = "";

            return;

        }

 

        // --- FORMULA RICHIESTA ---

        // Formula: (LC*4,5 – OP)/(LC*4,5)*80 + (LC*2 – PP)/(LC*2)*20

        const parte1 = ((LC * 4.5 - OP) / (LC * 4.5)) * 80;

        const parte2 = ((LC * 2 - PP) / (LC * 2)) * 20;

        const risultato = parte1 + parte2;

 

        // Visualizzazione percentuale

        spanPercentuale.innerText = risultato.toFixed(2) + "%";

 

        // Logica Output

        if (risultato >= 70) {

            divMessaggio.innerText = "Frequenza minima raggiunta";

            divMessaggio.className = "raggiunta";

        } else {

            // Ho aggiunto "NON" per distinguere l'esito negativo

            divMessaggio.innerText = "Frequenza minima NON raggiunta";

            divMessaggio.className = "non-raggiunta";

        }

    }

</script>
